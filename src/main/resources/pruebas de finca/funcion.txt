function generarMesas(seccion, numMesas) {
        for (let i = 0; i < numMesas; i++) {
            const tableWrapper = document.createElement('div');
            tableWrapper.classList.add('table-wrapper');

            // Crear la mesa
            const table = document.createElement('div');
            table.classList.add('table');
            tableWrapper.appendChild(table);

            // Crear las sillas alrededor de la mesa
            for (let j = 0; j < 4; j++) {
                const chair = document.createElement('div');
                chair.classList.add('chair');
                tableWrapper.appendChild(chair);
            }

            // Añadir la mesa a la sección
            seccion.appendChild(tableWrapper);
        }
    }

    // Generar mesas en las secciones General Izquierda y Derecha
    const generalIzquierda = document.querySelector('.general-izq');
    const generalDerecha = document.querySelector('.general-der');
    generarMesas(generalIzquierda, 15);
    generarMesas(generalDerecha, 15);
